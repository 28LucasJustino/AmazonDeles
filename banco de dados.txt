drop database eles;
CREATE DATABASE eles;
USE eles;

CREATE TABLE usuario(
idUsuario INT AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(100),
senha VARCHAR(45),
email VARCHAR(130),
cpf varchar(11),
telefone varchar(11));

CREATE TABLE produto(
idProduto INT AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(45),
categoria VARCHAR(40),
descricao tinytext,
preco float,
estoque int);
